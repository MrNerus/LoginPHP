
CREATE DATABASE `myusers`;
USE `myusers`;


CREATE TABLE `userdatas` (
  `u_id` int(11) NOT NULL,
  `fname` varchar(64) NOT NULL,
  `lname` varchar(64) NOT NULL,
  `email` varchar(64) NOT NULL,
  `phno` varchar(20) NOT NULL,
  `dob` date NOT NULL,
  `gender` varchar(5) NOT NULL
);


CREATE TABLE `userpasswords` (
  `u_id` int(11) NOT NULL,
  `password` varchar(128) NOT NULL
);


CREATE TABLE `users` (
  `id` int(11) NOT NULL,
  `username` varchar(32) NOT NULL
);


ALTER TABLE `userdatas`
  ADD KEY `fk_userdatas_u_id` (`u_id`);


ALTER TABLE `userpasswords`
  ADD KEY `fk_u_Id` (`u_id`);


ALTER TABLE `users`
  ADD PRIMARY KEY (`id`),
  ADD UNIQUE KEY `username` (`username`);


ALTER TABLE `users`
  MODIFY `id` int(11) NOT NULL AUTO_INCREMENT;



ALTER TABLE `userdatas`
  ADD CONSTRAINT `fk_userdatas_u_id` FOREIGN KEY (`u_id`) REFERENCES `users` (`id`);


ALTER TABLE `userpasswords`
  ADD CONSTRAINT `fk_u_Id` FOREIGN KEY (`u_id`) REFERENCES `users` (`id`);


CREATE TABLE userlogs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    u_id INT,
    logType VARCHAR(5) NOT NULL,
    logTimeStamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_users_u_id FOREIGN KEY (u_id) REFERENCES users(id)
);
